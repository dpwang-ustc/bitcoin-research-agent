Python API 使用示例 
from src.agent import BitcoinResearchAgent

# 创建 Agent
agent = BitcoinResearchAgent(
    llm_provider="openai",
    llm_model="gpt-4o-mini"
)

# 运行分析
result = agent.run("生成本周比特币市场完整分析报告")

# 获取报告
print(result['report'])

# 查看执行步骤
for msg in result['messages']:
    print(msg.content)
